# ============================================
# OPTION 1: OLLAMA (RECOMMANDÉ - 100% GRATUIT)
# ============================================
# Ollama fonctionne localement, aucune clé nécessaire !
# Installez Ollama depuis https://ollama.com
# Puis téléchargez un modèle : ollama pull llama3.2
# 
# Configuration optionnelle (si Ollama n'est pas sur localhost:11434) :
# VITE_OLLAMA_BASE_URL=http://localhost:11434
# VITE_OLLAMA_MODEL=llama3.2
#
# Voir SETUP_OLLAMA.md pour les instructions complètes

# ============================================
# OPTION 2: OPENAI (PAYANT)
# ============================================
# Obtenez votre clé sur: https://platform.openai.com/api-keys
# VITE_OPENAI_API_KEY=sk-votre-cle-openai-ici

# ============================================
# OPTION 3: GOOGLE GEMINI (GRATUIT AVEC QUOTA)
# ============================================
# Obtenez votre clé sur: https://makersuite.google.com/app/apikey
# VITE_GEMINI_API_KEY=votre-cle-gemini-ici

# ============================================
# PRIORITÉ D'UTILISATION
# ============================================
